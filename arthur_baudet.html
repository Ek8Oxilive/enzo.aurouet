1 Tests (if/elif/else) & Boucles (for/while)

Bash Tests conditionnels

# Syntaxe if / elif / else
if [ "$1" = "-h" ] || [ "$1" = "--help" ] ; then
    echo "Aide"
    exit 0
elif [ -z "$1" ] ; then       # -z = chaîne vide
    echo "Aucun argument"
    exit 1
else
    echo "Argument: $1"
fi

# Opérateurs de test courants
[ -z "$var" ]      # var est vide
[ -n "$var" ]      # var non vide
[ "$a" = "$b" ]    # égalité de chaînes
[ "$a" != "$b" ]   # différence
[ "$a" -eq "$b" ]  # égalité numérique
[ "$a" -gt "$b" ]  # supérieur (numérique)
[ "$a" -lt "$b" ]  # inférieur
[ -f fichier ]     # fichier existe
[ -d dossier ]     # dossier existe
[ ! condition ]    # négation

# Alternative: test
test "$LOG_LEVEL" = "INFO" && echo "info"

PowerShell Tests conditionnels

# Syntaxe if / elseif / else
if ($user.expires -eq "never") {
    New-LocalUser -Name $user.username -NoPassword
} elseif ($expires -lt (Get-Date)) {
    Write-Output "Expiré"
} else {
    New-LocalUser -Name $user.username `
        -AccountExpires $user.expires -NoPassword
}

# Opérateurs de comparaison
-eq    # égal
-ne    # différent
-gt    # supérieur
-lt    # inférieur
-ge    # supérieur ou égal
-le    # inférieur ou égal
-like  # pattern matching (* ?)
-match # regex

# Exemple avec -like (TP4)
if ((./timeme $pass) -like "Success*") {
    echo "Trouvé: $pass"
}

Bash Boucles

# for — itérer sur des arguments
for a in "$@" ; do
    echo $a
done

# for — séquences / ranges
for n in {0..9}-{0..9}-{0..9}-{0..9} ; do
    echo $n
done

# for — itérer sur des lettres
for l in {a..z} ; do
    echo $l
done

# while — lire un fichier ligne par ligne
while IFS="," read -r username group expires ; do
    echo "$username dans $group"
done < <(tail -n +2 users.csv)

# while — boucle infinie
while true ; do
    # ...
    break   # sortir de la boucle
done

PowerShell Boucles

# foreach — itérer sur un tableau
foreach ($i in $arg) {
    Write-Output $i
}

# foreach — itérer sur un objet/collection
$users = Import-CSV .\users.csv
foreach ($user in $users) {
    Write-Output $user.username
}

# for — boucle classique
for ($i = 1 ; $i -le $n ; $i++) {
    Write-Output "Itération $i"
}

# for — boucles imbriquées (brute force)
foreach ($a in 1..9) {
    foreach ($b in 1..9) {
        # ...
    }
}

# while — boucle infinie
while ($true) {
    # ...
    break   # sortir
}
2 Scripts avec arguments

Bash Arguments

# Variables spéciales
$0           # nom du script
$1, $2...    # arguments positionnels
$@           # tous les arguments (séparés)
$*           # tous les arguments (un seul mot)
$#           # nombre d'arguments
$?           # code retour dernière commande

# Usage pattern classique (TP1: echo.sh)
usage () {
    echo "$0 [-h|--help] arg [arg ...]"
}

if [ -z $1 ] ; then
    usage
    exit 1
fi

if [ $1 = "-h" ] || [ $1 = "--help" ] ; then
    usage
    exit 0
fi

# Itérer sur tous les arguments
for a in "$@" ; do
    echo $a
done
⚠️ exit 0 = succès, exit 1 = erreur

PowerShell Paramètres

# Bloc Param (en haut du script)
Param(
    [Parameter(Mandatory)]    # obligatoire
    [String[]]$arg            # tableau de chaînes
)

# Param avec valeur par défaut
Param(
    [int]$n,                  # optionnel, défaut 0
    [String]$lang             # optionnel, défaut ""
)

# Aide intégrée (Get-Help .\Script.ps1)
<#
    .SYNOPSIS
    Description courte du script.

    .PARAMETER arg
    Description du paramètre.

    .EXAMPLE
    PS> .\Script.ps1 hello
#>
Param(
    [Parameter(Mandatory)]
    [String[]]$arg
)

# Si Mandatory, PowerShell demande la
# valeur interactivement si non fournie
PowerShell gère automatiquement -h via Get-Help
3 Manipulation de chaînes — Bash (grep, cut, head, tail, awk…)

Bash grep — Chercher

# Compter les occurrences
grep -c 'File does not exist' Apache.log

# Expression régulière étendue
grep -E 'pattern1|pattern2' fichier

# Recherche récursive
grep -r 'Bad protocol' SSH.log

# Inverser la recherche
grep -v 'not' fichier

# Première occurrence seulement
grep -m 1 'model name' /proc/cpuinfo

# Capture regex (sed)
sed -nr "s/^.*epeated ([0-9]) times.*/\1/p" SSH.log

Bash cut — Découper

# Par délimiteur (espace par défaut, ici ':')
cut -d ':' -f 2 fichier    # 2e champ

# Par délimiteur espace
cut -d ' ' -f 3- fichier   # du 3e champ à la fin

# Par position de caractères
cut -c 6-11 fichier         # caractères 6 à 11
cut -c 1-6 fichier          # 6 premiers caractères

# Combinaison fréquente (TP2)
grep 'pattern' log | cut -d ':' -f 2

Bash head, tail, sort, uniq

# head — début du fichier
head -n 1 fichier         # 1ère ligne
sudo fdisk -l | head -n 1

# tail — fin / sauter des lignes
tail -n +2 fichier        # tout sauf la 1ère ligne
                          # (utile pour CSV: sauter l'en-tête)

# sort & uniq — trier et compter
sort fichier              # trier
sort -u fichier           # trier + unique
uniq -c                   # compter les répétitions

# Pipeline classique (TP2)
grep 'pattern' log | cut -c 1-6 | sort | uniq -c

Bash awk & autres

# awk — découpe flexible (espaces multiples)
awk '{print $2}' fichier       # 2e colonne
awk -F 'from ' '{print $2}'   # délimiteur custom
awk '{print $2/1024 " GiB"}'  # calculs

# paste — joindre des lignes
paste -sd + fichier   # concatène avec '+'

# Manipulation de chaînes bash
${var## }       # enlever espaces en début
${var%% }       # enlever espaces en fin

# command -v — vérifier si commande existe
if ! command -v lolcat &> /dev/null ; then
    echo "lolcat requis"
    exit 1
fi
4 Manipulation d'objets — PowerShell

PowerShell Accéder aux attributs

# Notation point pour accéder aux propriétés
$user.username
$user.group
$user.expires

# Objets retournés par cmdlets
$addr = Get-NetIPAddress -AddressFamily IPv4
$addr.IPAddress
$addr.PrefixLength

$mac = (Get-NetAdapter -Name $iface).MacAddress

# Interpolation dans les chaînes
Write-Output "$($addr.IPAddress)/$($addr.PrefixLength)"
# $() permet d'évaluer une expression dans ""

# Lister les propriétés d'un objet
Get-NetAdapter | Get-Member

PowerShell Filtrer & Pipeline

# Pipeline — passer des objets
Get-Process | Where-Object { $_.CPU -gt 10 }

# Import-CSV retourne des objets
$users = Import-CSV .\users.csv
# Chaque ligne = objet avec propriétés
# = noms de colonnes du CSV

# Filtrer avec Where-Object
$admins = $users | Where-Object {
    $_.group -eq "admin"
}

# ForEach-Object dans le pipeline
$users | ForEach-Object {
    Write-Output $_.username
}

# Select-Object — choisir des colonnes
Get-Process | Select-Object Name, CPU

# Measure-Command — mesurer le temps
$time = (Measure-Command {
    ./timeme "$pass$try"
}).TotalMilliseconds

PowerShell Cmdlets utiles (Get-CIMInstance)

# Informations système (TP3: Info.ps1)
# CPU
(Get-CIMInstance Win32_Processor).Name
(Get-CIMInstance Win32_Processor).NumberOfCores

# RAM
(Get-CIMInstance Win32_PhysicalMemory |
    Measure-Object Capacity -Sum).Sum / 1GB

# Disque
(Get-CIMInstance Win32_DiskDrive).Size / 1GB

# Hostname
$env:COMPUTERNAME
# ou: hostname

# Réseau
Get-NetIPAddress -AddressFamily IPv4
Get-NetAdapter -Name "Ethernet"

PowerShell Variables d'environnement

# Lire une variable d'environnement
$env:LOG_LEVEL
$env:COMPUTERNAME

# Définir avant l'appel du script
$env:LOG_LEVEL = "DEBUG"
.\Fcats.ps1

# Tester une variable d'environnement
if ($env:LOG_LEVEL -eq "DEBUG") {
    Write-Output "[DEBUG] message"
}

# Web requests (TP3/TP4)
$response = Invoke-WebRequest -Uri $URL `
    -UseBasicParsing
$data = ConvertFrom-Json $response.Content
$data.fact        # accéder à une propriété JSON
5 Gestion des utilisateurs & groupes

Bash Utilisateurs & Groupes

# Créer un groupe (TP2: create-users.sh)
groupadd -f mongroupe    # -f = pas d'erreur si existe

# Créer un utilisateur
useradd -m username -g groupe
#   -m = créer le répertoire HOME
#   -g = groupe principal

# Avec date d'expiration
useradd -m username -g groupe -e 2030-06-09

# Supprimer un utilisateur
userdel -r username      # -r = supprimer le HOME

# Supprimer un groupe
groupdel -f mongroupe

# Lecture CSV pour création (pattern complet)
while IFS="," read -r username group expires ; do
    groupadd -f $group
    if [ $expires = "never" ] ; then
        useradd -m $username -g $group
    else
        useradd -m $username -g $group -e $expires
    fi
done < <(tail -n +2 users.csv)
tail -n +2 permet de sauter la ligne d'en-tête du CSV

PowerShell Utilisateurs & Groupes

# Créer un groupe local
New-LocalGroup -Name "admin"

# Créer un utilisateur local
New-LocalUser -Name "TobiasForge" -NoPassword

# Avec date d'expiration
New-LocalUser -Name "MadGallica" `
    -AccountExpires "2030-06-09" -NoPassword

# Ajouter au groupe
Add-LocalGroupMember -Group "admin" `
    -Member "TobiasForge"

# Supprimer
Remove-LocalUser TobiasForge, PerEriksson
Remove-LocalGroup admin, ghost

# Pattern complet avec Import-CSV (TP4)
$users = Import-CSV .\users.csv

foreach ($user in $users) {
    New-LocalGroup -Name $user.group
    if ($user.expires -eq "never") {
        New-LocalUser -Name $user.username `
            -NoPassword
    } else {
        New-LocalUser -Name $user.username `
            -AccountExpires $user.expires `
            -NoPassword
    }
    Add-LocalGroupMember -Group $user.group `
        -Member $user.username
}
⚠️ Pensez à tester que le fichier CSV existe avant de le lire ! Bash : if [ ! -f users.csv ] ; then echo "requis"; exit 1; fi — PowerShell : if (-not (Test-Path .\users.csv)) { exit 1 }
6 Formats de sérialisation — JSON & CSV

Bash CSV avec read + IFS

# Lire un CSV ligne par ligne
while IFS="," read -r col1 col2 col3 ; do
    echo "Col1=$col1 Col2=$col2 Col3=$col3"
done < <(tail -n +2 fichier.csv)
# tail -n +2 : saute l'en-tête

# IFS = Internal Field Separator
# IFS="," indique que les champs sont
# séparés par des virgules

Bash JSON avec jq

# Afficher du JSON formaté
cat data.json | jq .

# Accéder à un champ
echo $json | jq .ifname
echo $json | jq .addr_info[0].local

# Premier élément d'une liste
echo $json | jq .[0]

# Filtrer avec select
echo $json | jq '.[] | select(.ifname == "eth0")'

# -r : sortie brute (sans guillemets)
echo $json | jq -r .address

# Exemple réel (TP2: ip.sh)
ip_data=$(ip -j a)
ifname_data=$(echo $ip_data | \
    jq ".[] | select(.ifname == \"$1\")")
echo $ifname_data | jq -r .address

PowerShell CSV avec Import-CSV

# Lire un CSV — retourne des objets
$users = Import-CSV .\users.csv

# Chaque ligne devient un objet avec
# les en-têtes comme propriétés
foreach ($user in $users) {
    $user.username   # colonne "username"
    $user.group      # colonne "group"
    $user.expires    # colonne "expires"
}

# Exporter en CSV
$data | Export-CSV .\sortie.csv -NoTypeInfo

PowerShell JSON

# Parser du JSON depuis une API
$response = Invoke-WebRequest -Uri $URL `
    -UseBasicParsing
$data = ConvertFrom-Json $response.Content
$data.fact      # accéder à une propriété

# Convertir un objet en JSON
$obj | ConvertTo-Json

# Convertir une chaîne JSON en objet
$obj = '{"name":"test"}' | ConvertFrom-Json
$obj.name   # "test"
7 Utiliser des outils non encore connus

🔍 Méthodologie

Le DS peut vous demander d'utiliser une commande ou un outil que vous n'avez jamais vu. Voici l'approche à suivre : 

# 1. Lire l'aide de la commande
commande --help
commande -h
man commande          # page de manuel complète

# 2. Identifier les options utiles
# Chercher les patterns : -o output, -f file, -v verbose...

# 3. Tester avec un exemple simple
# Toujours essayer un cas basique avant de scripter

# 4. Adapter l'exemple à votre besoin
# Combiner avec les outils que vous connaissez (pipe, grep...)
# PowerShell — Découvrir une cmdlet
Get-Help NomCmdlet            # aide
Get-Help NomCmdlet -Examples  # exemples
Get-Help NomCmdlet -Full      # aide complète

# Lister les propriétés d'un objet retourné
NomCmdlet | Get-Member

# Chercher une cmdlet par mot-clé
Get-Command *keyword*
Exemples vus en TP : jq (JSON en Bash), Get-CIMInstance (info système), Invoke-WebRequest (API), Measure-Command (timing). Le principe est toujours le même : lire la doc, adapter un exemple.
⚡ Patterns récurrents des TP — à connaître par cœur

Bash Vérifier un fichier + lire CSV

#!/bin/bash
FILE=users.csv

if [ ! -f $FILE ] ; then
    echo "$FILE required"
    exit 1
fi

while IFS="," read -r user group exp ; do
    echo "$user / $group / $exp"
done < <(tail -n +2 $FILE)

PowerShell Lire CSV + agir

$users = Import-CSV .\users.csv

foreach ($u in $users) {
    if ($u.expires -eq "never") {
        # action sans expiration
    } else {
        # action avec expiration
    }
}

Bash Analyse de logs

# Compter un pattern
grep -c 'pattern' fichier.log

# Compter par jour (6 premiers chars = date)
grep 'pattern' log | cut -c 1-6 | \
    sort | uniq -c

# Extraire un champ avec awk
grep 'pattern' log | \
    awk -F 'from ' '{print $2}' | \
    awk -F ' port' '{print $1}' | \
    sort | uniq -c

Bash Info système

# Hostname
cat /etc/hostname
# ou: $HOSTNAME

# CPU
grep -m 1 'model name' /proc/cpuinfo | \
    cut -d ' ' -f 3-
grep -c 'model name' /proc/cpuinfo

# RAM
grep MemTotal /proc/meminfo | \
    awk '{print $2/1024/1024 " GiB"}'

# Ou avec lscpu, free -h, lsblk...

ip_data=$(ip -j a)
